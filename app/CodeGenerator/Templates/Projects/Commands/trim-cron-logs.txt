<?php

namespace App\Console\Commands;

use Illuminate\Console\Command;
use App\Services\Crons\CronService;

class TrimCronLogs extends Command
{
    /**
     * The name and signature of the console command.
     *
     * @var string
     */
    protected $signature = 'app:trim-cron-logs';

    /**
     * The console command description.
     *
     * @var string
     */
    protected $description = 'Trim Logs older than a week';

    /**
     * Execute the console command.
     */

    public function handle()
    {

        $interval = 'everyMinute';
        
        CronService::runAndLogCron($this->signature, $this->description, 'TrimCronLogsHandler', 'handleTrimCronLogs', $interval);

    }

    

}
