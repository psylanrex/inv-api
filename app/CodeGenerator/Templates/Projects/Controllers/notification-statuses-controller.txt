<?php

namespace App\Http\Controllers\Admin\NotificationStatuses;

use App\Http\Controllers\Controller;
use Illuminate\Http\Request;
use Illuminate\Validation\Rule;
use App\Models\NotificationStatus;
use App\Utilities\DropDownListMaker;
use App\Utilities\SortBy;
use App\Services\Forms\AddFormConfigService;
use App\Services\Forms\EditFormConfigService;

class NotificationStatusesController extends Controller
{

    public $indexColumnMap = ['id', 'notification_status_name'];

    public function index(Request $request)
    {
        
        $column_map = $this->indexColumnMap;

        list($sortBy, $sortOrder) = SortBy::setSortBy($request, $column_map);
        
        $notificationStatuses = NotificationStatus::select($column_map)     
                                        ->orderBy($sortBy, $sortOrder)                                                  
                                        ->paginate(10);

        return $notificationStatuses;

    }

    /**
     * Store a newly created resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @return \Illuminate\Http\Response
     */
    public function store(Request $request)
    {

           $request->validate([
   
               'notification_status_name' => 'string|max:140|required|unique:notification_statuses,notification_status_name',
                               
           ]);
       
           return NotificationStatus::create($request->all());

    }

    /**
     * Display the specified resource.
     *
     * @param  int  $id
     * @return \Illuminate\Http\Response
     */
    public function show($id)
    {
        
        return NotificationStatus::select('id', 'notification_status_name', 'created_at', 'updated_at')->where('id', $id)->first();
      
    }

    public function addFormConfig(AddFormConfigService $service, DropDownListMaker $listMaker)
    {

       return $service->getFormConfig('NotificationStatus', 'notification_statuses', $listMaker);


    }

    public function editFormConfig($id, EditFormConfigService $service, DropDownListMaker $listMaker)
    { 

        return $service->getFormConfig($id, 'NotificationStatus', 'notification_statuses', $listMaker); 

    }

    /**
     * Update the specified resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @param  int  $id
     * @return \Illuminate\Http\Response
     */
    public function update(Request $request, $id)
    {

        $request->validate([

            'notification_status_name' => [
                'required',
                'string',
                'max:140',

                Rule::unique('notification_statuses')->ignore($id),
            ]

        ]);

        $notificationStatus = NotificationStatus::find($id);

        $notificationStatus->update($request->all());

        return $notificationStatus;

    }

    /**
     * Remove the specified resource from storage.
     *
     * @param  int  $id
     * @return \Illuminate\Http\Response
     */
    public function destroy($id)
    {
        
        return NotificationStatus::destroy($id);

    }

    /**
     * Find the specified resource.
     *
     * @param  string  $name
     * @return \Illuminate\Http\Response
     */
    public function search(Request $request, $keyword)
    {
        $column_map = $this->indexColumnMap;

        list($sortBy, $sortOrder) = SortBy::setSortBy($request, $column_map);
        
        $notificationStatuss = NotificationStatus::select($column_map)
                                            ->where('notification_status_name', 'like', '%'.$keyword.'%')
                                            ->orderBy($sortBy, $sortOrder)
                                            ->paginate(10);

        return $notificationStatuss;

    }

}
