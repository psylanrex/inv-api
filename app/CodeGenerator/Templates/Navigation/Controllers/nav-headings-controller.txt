<?php

namespace App\Http\Controllers\Admin\NavHeadings;

use App\Http\Controllers\Controller;
use Illuminate\Http\Request;
use Illuminate\Validation\Rule;
use App\Models\NavHeading;
use App\Utilities\DropDownListMaker;
use App\Utilities\SortBy;
use App\Services\Forms\AddFormConfigService;
use App\Services\Forms\EditFormConfigService;

class NavHeadingsController extends Controller
{

    public $indexColumnMap = ['id', 'nav_heading_name'];

    public function getNavHeadingsList()
    {

        return NavHeading::select('id as nav_heading_id', 'nav_heading_name')->get();

    }

    public function index(Request $request)
    {
        
        $column_map = $this->indexColumnMap;

        list($sortBy, $sortOrder) = SortBy::setSortBy($request, $column_map);
        
        $navHeadings = NavHeading::select($column_map)     
                                   ->orderBy($sortBy, $sortOrder)                                                  
                                   ->paginate(10);

        return $navHeadings;

    }

    /**
     * Store a newly created resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @return \Illuminate\Http\Response
     */
    public function store(Request $request)
    {

           $request->validate([
   
               'nav_heading_name' => 'string|max:140|required|unique:nav_headings,nav_heading_name',
                               
           ]);
       
           return NavHeading::create($request->all());

    }

    /**
     * Display the specified resource.
     *
     * @param  int  $id
     * @return \Illuminate\Http\Response
     */
    public function show($id)
    {
        
        return NavHeading::select('id', 'nav_heading_name', 'created_at', 'updated_at')->where('id', $id)->first();
      
    }

    public function addFormConfig(AddFormConfigService $service, DropDownListMaker $listMaker)
    {

       return $service->getFormConfig('NavHeading', 'nav_headings', $listMaker);


    }

    public function editFormConfig($id, EditFormConfigService $service, DropDownListMaker $listMaker)
    { 

        return $service->getFormConfig($id, 'NavHeading', 'nav_headings', $listMaker); 

    }

    /**
     * Update the specified resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @param  int  $id
     * @return \Illuminate\Http\Response
     */
    public function update(Request $request, $id)
    {

        $request->validate([

            'nav_heading_name' => [
                'required',
                'string',
                'max:140',

                Rule::unique('nav_headings')->ignore($id),
            ]

        ]);

        $navHeading = NavHeading::find($id);

        $navHeading->update($request->all());

        return $navHeading;

    }

    /**
     * Remove the specified resource from storage.
     *
     * @param  int  $id
     * @return \Illuminate\Http\Response
     */
    public function destroy($id)
    {
        
        return NavHeading::destroy($id);

    }

    /**
     * Find the specified resource.
     *
     * @param  string  $name
     * @return \Illuminate\Http\Response
     */
    public function search(Request $request, $keyword)
    {
        $column_map = $this->indexColumnMap;

        list($sortBy, $sortOrder) = SortBy::setSortBy($request, $column_map);
        
        $navHeadings = NavHeading::select($column_map)
                                   ->where('nav_heading_name', 'like', '%'.$keyword.'%')
                                   ->orderBy($sortBy, $sortOrder)
                                   ->paginate(10);

        return $navHeadings;

    }

}
