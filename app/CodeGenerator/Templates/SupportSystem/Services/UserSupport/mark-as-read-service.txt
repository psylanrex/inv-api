<?php

namespace App\Services\Support\UserSupport;

use App\Models\TicketResponse;

class MarkAsReadService
{

    public static function markRead($request)
    {

        $user_id = auth()->id();

        $ticket_response_id = $request->get('ticket_response_id');

        
        $response = TicketResponse::find($ticket_response_id);

        $response->is_read = 1;

        $response->save();

        return ['message' => 'The ticket response has been read'];

    }

}
