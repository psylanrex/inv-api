<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Factories\HasFactory;
use Illuminate\Database\Eloquent\Model;
use Carbon\Carbon;

class FormConfig extends Model
{
    use HasFactory;

    protected $fillable = [

        'crud_id',
        'name',
        'type',
        'label',
        'required',
        'max_length',
        'default_value',
        'instructions'

    ];

    protected $casts = [
        'created_at' => 'date:Y-m-d',
        'updated_at' => 'date:Y-m-d'
    ];
}
