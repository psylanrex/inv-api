<?php

namespace App\Http\Controllers\User\Contents;

use App\Http\Controllers\Controller;
use Illuminate\Http\Request;
use App\Models\PageContent;

class ContentsController extends Controller
{
    public function terms()
    {

        $content = PageContent::where('page_content_name', 'terms')->first();

        return $content->page_content_body;

    }

    public function privacy()
    {

        $content = PageContent::where('page_content_name', 'privacy')->first();

        return $content->page_content_body;

    }

    public function about()
    {

        $content = PageContent::where('page_content_name', 'about')->first();

        return $content->page_content_body;

    }
}
